import React, { useState, useEffect, useRef } from 'react';

import { 

  Save, 

  History, 

  Plus, 

  Trash2, 

  TrendingDown, 

  Calendar,

  Camera,

  Loader2, 

  CheckCircle2, 

  Settings, 

  X, 

  Sunrise, 

  Sunset, 

  Flame,

  Utensils,

  BarChart3,

  PieChart,

  Pencil,

  Check,

  Cloud,

  CloudOff,

  RefreshCw,

  AlertTriangle

} from 'lucide-react';


const App = () => {

  const [activeTab, setActiveTab] = useState('record');

  const [history, setHistory] = useState([]);

  const [isAnalyzing, setIsAnalyzing] = useState(null); 

  const [isSyncing, setIsSyncing] = useState(false); 

  const [isLoadingHistory, setIsLoadingHistory] = useState(false);

  const [showSettings, setShowSettings] = useState(false);

  const [cloudError, setCloudError] = useState(false);

  

  const [editingId, setEditingId] = useState(null);

  const [editBuffer, setEditBuffer] = useState(null);


  const STORAGE_KEY = 'kebab_stock_v18_history';

  const DEFAULT_SHEET_URL = 'https://script.google.com/macros/s/AKfycby46zKnV7tYwb3IHVQiHhEo31XY2rRZCwOyQU-dM60g-yIcA-KWliXDMPGidZ7p8KZO5g/exec';

  const [sheetUrl, setSheetUrl] = useState(DEFAULT_SHEET_URL);

  

  const fileInputRef = useRef(null);

  const [pendingTarget, setPendingTarget] = useState(null);


  const today = new Date().toISOString().split('T')[0];

  const [selectedDate, setSelectedDate] = useState(today);


  const apiKey = ""; 


  const itemsList = [

    { id: 'chicken', label: 'ไก่', unit: 'กรัม', cat: 'meat' },

    { id: 'carrot', label: 'แครอท', unit: 'กรัม', cat: 'veg' },

    { id: 'cabbage', label: 'กะหล่ำ', unit: 'กรัม', cat: 'veg' },

    { id: 'lettuce', label: 'ผักกาด', unit: 'กรัม', cat: 'veg' },

    { id: 'cheese', label: 'ชีส', unit: 'กรัม', cat: 'topping' },

    { id: 'regularWrap', label: 'แป้งธรรมดา', unit: 'แผ่น', cat: 'wrap' },

    { id: 'miniWrap', label: 'แป้งมินิ', unit: 'แผ่น', cat: 'wrap' },

    { id: 'gas', label: 'แก๊ส', unit: 'กก.', cat: 'utility' },

  ];


  const salesTypes = [

    { id: 'soft', label: 'ไก่นุ่ม' },

    { id: 'cheese_kebab', label: 'ไก่ชีส' },

    { id: 'crispy', label: 'ไก่กรอบ' },

    { id: 'crispy_cheese', label: 'ไก่กรอบชีส' },

  ];


  const catColors = {

    meat: "bg-rose-50 text-rose-700 border-rose-100 bar:bg-rose-500",

    veg: "bg-emerald-50 text-emerald-700 border-emerald-100 bar:bg-emerald-500",

    topping: "bg-amber-50 text-amber-700 border-amber-100 bar:bg-amber-500",

    wrap: "bg-sky-50 text-sky-700 border-sky-100 bar:bg-sky-500",

    utility: "bg-orange-50 text-orange-700 border-orange-100 bar:bg-orange-500"

  };


  const getInitialState = () => {

    const stock = {};

    itemsList.forEach(item => { stock[item.id] = { before: '', after: '' }; });

    const sales = {};

    salesTypes.forEach(type => { sales[type.id] = ''; });

    return { stock, sales };

  };


  const [currentData, setCurrentData] = useState(getInitialState());

  const [sessionStatus, setSessionStatus] = useState('morning'); 

  const [note, setNote] = useState('');


  const formatDisplayDate = (dateStr) => {

    try {

      const d = new Date(dateStr);

      if (!isNaN(d.getTime())) {

        return d.toLocaleDateString('th-TH', { year: 'numeric', month: 'short', day: 'numeric' });

      }

    } catch(e){}

    return dateStr;

  };


  const fetchCloudHistory = async () => {

    if (!sheetUrl) return;

    setIsLoadingHistory(true);

    setCloudError(false);

    

    const cleanUrl = sheetUrl.trim().split(' ')[0].replace(/\.$/, '');

    

    try {

      const controller = new AbortController();

      const timeoutId = setTimeout(() => controller.abort(), 10000);


      const response = await fetch(cleanUrl, { 

        method: 'GET',

        signal: controller.signal,

        headers: {

          'Accept': 'application/json'

        }

      });

      

      clearTimeout(timeoutId);


      const contentType = response.headers.get("content-type");

      

      if (contentType && contentType.includes("application/json")) {

        const data = await response.json();

        if (Array.isArray(data)) {

          // ซ่อนข้อมูลที่ซ้ำซ้อนโดยยึดวันที่ล่าสุดเป็นหลัก (Deduplication)

          const uniqueData = [];

          const seenDates = new Set();

          for (const item of data) {

            if (!seenDates.has(item.date)) {

              seenDates.add(item.date);

              uniqueData.push(item);

            }

          }

          

          const sortedData = uniqueData.sort((a, b) => new Date(b.date) - new Date(a.date));

          setHistory(sortedData);

          localStorage.setItem(STORAGE_KEY, JSON.stringify(sortedData));

          setCloudError(false);

        } else if (data.error) {

          setCloudError(true);

        }

      } else {

        setCloudError(true);

      }

    } catch (e) {

      // จัดการข้อผิดพลาดเงียบๆ เพื่อไม่ให้แอปแครชจากปัญหา Network หรือ CORS

      setCloudError(true);

    } finally {

      setIsLoadingHistory(false);

    }

  };


  useEffect(() => {

    const localHistory = localStorage.getItem(STORAGE_KEY);

    if (localHistory) setHistory(JSON.parse(localHistory));

    

    const savedUrl = localStorage.getItem('kebab_sheet_url_v16');

    if (savedUrl) setSheetUrl(savedUrl);


    const activeSession = localStorage.getItem('kebab_active_session_v16');

    if (activeSession) {

      const parsed = JSON.parse(activeSession);

      setCurrentData(parsed.data || getInitialState());

      setSessionStatus(parsed.status || 'morning');

      setNote(parsed.note || '');

      if (parsed.selectedDate) setSelectedDate(parsed.selectedDate);

    }


    fetchCloudHistory();

  }, []);


  useEffect(() => {

    if (activeTab === 'record') {

      localStorage.setItem('kebab_active_session_v16', JSON.stringify({

        data: currentData, status: sessionStatus, note, selectedDate

      }));

    }

    if (activeTab === 'history') {

      fetchCloudHistory();

    }

  }, [activeTab]);


  const syncToSheet = async (record) => {

    if (!sheetUrl) return;

    const cleanUrl = sheetUrl.trim().split(' ')[0].replace(/\.$/, '');

    setIsSyncing(true);

    try {

      await fetch(cleanUrl, { 

        method: 'POST', 

        mode: 'no-cors', 

        cache: 'no-cache',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify(record) 

      });

      setTimeout(fetchCloudHistory, 2000); 

    } catch (e) {

      setCloudError(true);

    } finally {

      setIsSyncing(false);

    }

  };


  const saveEdit = async () => {

    const totalSales = Object.keys(editBuffer.sales)

      .filter(key => key !== 'total')

      .reduce((sum, key) => sum + (parseInt(editBuffer.sales[key]) || 0), 0);

    

    const finalEdit = { ...editBuffer, sales: { ...editBuffer.sales, total: totalSales }, synced: false };


    setEditingId(null);

    setEditBuffer(null);

    

    await syncToSheet(finalEdit);

  };


  const deleteRecord = async (record) => {

    if (!confirm(`ยืนยันการลบประวัติวันที่ ${formatDisplayDate(record.date)}?\n\n(ข้อมูลนี้จะถูกลบออกจาก Google Sheet ทันที)`)) return;

    

    // ลบออกจากหน้าจอและ LocalStorage ทันที

    const newHistory = history.filter(h => h.id !== record.id);

    setHistory(newHistory);

    localStorage.setItem(STORAGE_KEY, JSON.stringify(newHistory));

    

    if (!sheetUrl) return;

    const cleanUrl = sheetUrl.trim().split(' ')[0].replace(/\.$/, '');

    const postUrl = cleanUrl + (cleanUrl.includes('?') ? '&' : '?') + 't=' + Date.now();

    setIsSyncing(true);

    

    try {

      await fetch(postUrl, {

        method: 'POST',

        mode: 'no-cors',

        headers: { 'Content-Type': 'text/plain' },

        body: JSON.stringify({ action: 'delete', date: record.date, id: record.id })

      });

      // หน่วงเวลารอ Sheet ลบให้เสร็จก่อนดึงใหม่

      setTimeout(fetchCloudHistory, 2500);

    } catch (e) {

      console.error("Delete sync error:", e);

      setCloudError(true);

    } finally {

      setIsSyncing(false);

    }

  };


  const handleFinishDay = async () => {

    const hasData = itemsList.some(item => currentData.stock[item.id].before !== '' || currentData.stock[item.id].after !== '');

    if (!hasData) { alert("กรุณากรอกข้อมูล"); return; }

    

    const newRecord = {

      id: Date.now(),

      rawDate: selectedDate,

      date: selectedDate, // ใช้ YYYY-MM-DD เพื่อป้องกัน Google แปลงค่าวันที่ผิดพลาด

      stock: { ...currentData.stock },

      sales: { ...currentData.sales, total: calculateTotalSales() },

      note,

      synced: false

    };

    

    // อัปเดตประวัติในหน้าจอทันที และเคลียร์รายการซ้ำถ้ามีวันที่เดียวกัน

    const filteredHistory = history.filter(h => h.date !== selectedDate);

    const updatedHistory = [newRecord, ...filteredHistory].sort((a, b) => new Date(b.date) - new Date(a.date));

    

    setHistory(updatedHistory);

    localStorage.setItem(STORAGE_KEY, JSON.stringify(updatedHistory));

    

    setCurrentData(getInitialState());

    setSessionStatus('morning');

    setNote('');

    setSelectedDate(today);

    localStorage.removeItem('kebab_active_session_v16');

    

    await syncToSheet(newRecord);

    setActiveTab('history');

  };


  const calculateTotalSales = () => {

    return Object.values(currentData.sales).reduce((sum, val) => sum + (parseInt(val) || 0), 0);

  };


  return (

    <div className="min-h-screen bg-slate-50 font-sans text-slate-900 pb-16">

      <input type="file" accept="image/*" capture="environment" ref={fileInputRef} className="hidden" 

        onChange={(e) => {

          const file = e.target.files[0];

          if (!file || !pendingTarget) return;

          const reader = new FileReader();

          reader.onloadend = async () => {

            const base64Data = reader.result.split(',')[1];

            setIsAnalyzing({ id: pendingTarget.id, type: pendingTarget.type });

            try {

              const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                method: 'POST', headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify({ contents: [{ role: "user", parts: [{ text: "Extract only the numeric value. Return ONLY the number." }, { inlineData: { mimeType: "image/png", data: base64Data } }] }] })

              });

              const result = await response.json();

              const cleaned = result.candidates?.[0]?.content?.parts?.[0]?.text?.replace(/[^0-9.]/g, '') || '';

              if (cleaned) {

                setCurrentData(prev => ({ ...prev, stock: { ...prev.stock, [pendingTarget.id]: { ...prev.stock[pendingTarget.id], [pendingTarget.type]: cleaned } } }));

              }

            } catch (error) {

              console.error(error);

            }

            setIsAnalyzing(null);

          };

          reader.readAsDataURL(file);

        }}

      />


      {showSettings && (

        <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-black/40 backdrop-blur-sm">

          <div className="bg-white w-full max-w-xs rounded-2xl p-5 shadow-2xl">

            <h2 className="text-sm font-black flex items-center gap-2 mb-3"><Settings size={16} /> ตั้งค่า Cloud</h2>

            <input type="text" value={sheetUrl} onChange={(e) => setSheetUrl(e.target.value)} className="w-full bg-slate-100 border rounded-lg px-3 py-2 text-[10px] mb-4 outline-none" />

            <button onClick={() => { localStorage.setItem('kebab_sheet_url_v16', sheetUrl.trim()); setShowSettings(false); fetchCloudHistory(); }} className="w-full bg-slate-900 text-white font-bold py-2 rounded-lg text-xs">บันทึกและโหลดใหม่</button>

          </div>

        </div>

      )}


      {isSyncing && <div className="fixed top-0 left-0 w-full h-1 z-50 bg-orange-100"><div className="h-full bg-orange-500 animate-pulse w-full"></div></div>}


      <header className="bg-slate-900 text-white px-3 py-2.5 flex justify-between items-center sticky top-0 z-30 shadow-sm border-b border-white/5">

        <h1 className="text-sm font-black flex items-center gap-1.5 italic tracking-tight"><Flame className="text-orange-500" size={16} /> KEBAB AI</h1>

        <div className="flex items-center gap-2">

          {isLoadingHistory ? (

            <Loader2 size={14} className="animate-spin text-blue-400" />

          ) : cloudError ? (

            <CloudOff size={14} className="text-rose-400" title="ออฟไลน์ / การเชื่อมต่อมีปัญหา" />

          ) : (

            <Cloud size={14} className="text-emerald-400" title="ออนไลน์" />

          )}

          <button onClick={() => setShowSettings(true)} className="p-1.5 bg-white/10 rounded-lg active:scale-95"><Settings size={14} /></button>

        </div>

      </header>


      <main className="max-w-md mx-auto p-2">

        <div className="flex bg-white rounded-lg shadow-sm mb-2 p-0.5 border border-slate-200">

          <button onClick={() => setActiveTab('record')} className={`flex-1 py-1.5 rounded-md text-[11px] font-bold transition-all ${activeTab === 'record' ? 'bg-orange-500 text-white shadow-sm' : 'text-slate-500'}`}>บันทึกสต็อก</button>

          <button onClick={() => setActiveTab('history')} className={`flex-1 py-1.5 rounded-md text-[11px] font-bold transition-all ${activeTab === 'history' ? 'bg-orange-500 text-white shadow-sm' : 'text-slate-500'}`}>ดูประวัติจาก Sheet</button>

        </div>


        {activeTab === 'record' ? (

          <div className="space-y-1.5 animate-in fade-in">

            <div className="bg-white px-3 py-2 rounded-xl border border-slate-200 flex items-center justify-between shadow-sm">

              <div className="flex items-center gap-2 font-black text-[10px] text-slate-700 uppercase"><Calendar size={14} className="text-orange-500" /> วันที่:</div>

              <input type="date" value={selectedDate} onChange={(e) => setSelectedDate(e.target.value)} className="bg-slate-100 rounded-lg px-2 py-1 text-[11px] font-bold outline-none" />

            </div>


            <div className={`px-2 py-1.5 rounded-xl border ${sessionStatus === 'morning' ? 'bg-blue-50 border-blue-100 text-blue-700' : 'bg-emerald-50 border-emerald-100 text-emerald-700'} flex justify-between items-center`}>

              <div className="flex items-center gap-2">{sessionStatus === 'morning' ? <Sunrise size={14}/> : <Sunset size={14}/>}<span className="text-[9px] font-black uppercase">{sessionStatus === 'morning' ? 'สต็อกเช้า' : 'สต็อกเย็น & ยอดขาย'}</span></div>

              <button onClick={() => setSessionStatus(sessionStatus === 'morning' ? 'evening' : 'morning')} className="text-[8px] font-black underline uppercase">สลับ</button>

            </div>


            <div className="bg-white rounded-xl shadow-sm border border-slate-200 overflow-hidden">

              <div className="grid grid-cols-12 bg-slate-100 py-1 px-2 border-b text-[8px] font-black text-slate-500 uppercase">

                <div className="col-span-4">รายการ</div><div className="col-span-4 text-center">เริ่ม</div><div className="col-span-4 text-center">เหลือ</div>

              </div>

              {itemsList.map((item) => (

                <div key={item.id} className="grid grid-cols-12 items-center gap-1 border-b last:border-0 py-1.5 px-2 hover:bg-slate-50/50">

                  <div className="col-span-4 flex items-center gap-1.5">

                    <div className={`w-1.5 h-1.5 rounded-full ${catColors[item.cat].split(' ')[0]}`}></div>

                    <div className="flex flex-col"><span className={`text-[10px] font-black leading-tight ${catColors[item.cat].split(' ')[1]}`}>{item.label}</span><span className="text-[7px] text-slate-400 font-bold uppercase">{item.unit}</span></div>

                  </div>

                  <div className="col-span-4 relative">

                    <input type="number" disabled={sessionStatus === 'evening'} value={currentData.stock[item.id].before} onChange={(e) => setCurrentData(p => ({...p, stock: {...p.stock, [item.id]: {...p.stock[item.id], before: e.target.value}}}))} className="w-full border border-slate-200 rounded-md pl-1 pr-6 py-1 text-[11px] font-bold text-center outline-none" />

                    {sessionStatus === 'morning' && <button onClick={() => {setPendingTarget({id: item.id, type: 'before'}); fileInputRef.current.click();}} className="absolute right-0.5 top-1/2 -translate-y-1/2 text-blue-500 p-0.5">{isAnalyzing?.id === item.id && isAnalyzing?.type === 'before' ? <Loader2 size={11} className="animate-spin" /> : <Camera size={11}/>}</button>}

                  </div>

                  <div className="col-span-4 relative">

                    <input type="number" disabled={sessionStatus === 'morning'} value={currentData.stock[item.id].after} onChange={(e) => setCurrentData(p => ({...p, stock: {...p.stock, [item.id]: {...p.stock[item.id], after: e.target.value}}}))} className="w-full border border-slate-200 rounded-md pl-1 pr-6 py-1 text-[11px] font-bold text-center outline-none" />

                    {sessionStatus === 'evening' && <button onClick={() => {setPendingTarget({id: item.id, type: 'after'}); fileInputRef.current.click();}} className="absolute right-0.5 top-1/2 -translate-y-1/2 text-emerald-500 p-0.5">{isAnalyzing?.id === item.id && isAnalyzing?.type === 'after' ? <Loader2 size={11} className="animate-spin" /> : <Camera size={11}/>}</button>}

                  </div>

                </div>

              ))}

            </div>


            {sessionStatus === 'evening' && (

              <div className="bg-white rounded-xl shadow-sm border border-slate-200 p-2 space-y-2 animate-in slide-in-from-top-2">

                <div className="flex items-center gap-2 mb-1 border-b pb-1 font-black text-[10px] text-slate-700 uppercase"><BarChart3 size={14}/> บันทึกยอดขาย (ชิ้น)</div>

                <div className="grid grid-cols-2 gap-2">

                  {salesTypes.map(type => (

                    <div key={type.id} className="flex flex-col gap-0.5">

                      <label className="text-[9px] font-bold text-slate-500 ml-1">{type.label}</label>

                      <input type="number" value={currentData.sales[type.id]} onChange={(e) => setCurrentData(p => ({...p, sales: {...p.sales, [type.id]: e.target.value}}))} className="w-full bg-slate-50 border border-slate-200 rounded-lg px-2 py-1 text-[11px] font-black focus:ring-1 focus:ring-orange-400 outline-none" />

                    </div>

                  ))}

                  <div className="col-span-2 bg-orange-50 rounded-lg p-2 flex justify-between items-center text-orange-700 font-black text-[10px]"><span>ยอดขายรวม:</span><span className="text-sm text-orange-600">{calculateTotalSales()} ชิ้น</span></div>

                </div>

              </div>

            )}

            

            <textarea value={note} onChange={(e) => setNote(e.target.value)} className="w-full bg-white border border-slate-200 rounded-xl p-2 text-[10px] font-bold h-12 outline-none focus:ring-1 focus:ring-slate-300" placeholder="หมายเหตุ..." />

            

            <button onClick={sessionStatus === 'morning' ? () => setSessionStatus('evening') : handleFinishDay} className={`w-full font-black py-2.5 rounded-xl shadow-md flex items-center justify-center gap-2 active:scale-95 text-xs text-white ${sessionStatus === 'morning' ? 'bg-slate-800' : 'bg-orange-600'}`}>

              <Save size={16} /> {sessionStatus === 'morning' ? 'ยืนยันสต็อกเช้า' : 'บันทึกปิดวันลง Google Sheet'}

            </button>

          </div>

        ) : (

          <div className="space-y-3 animate-in fade-in">

            <button onClick={fetchCloudHistory} disabled={isLoadingHistory} className="w-full py-2 bg-blue-50 text-blue-600 rounded-xl border border-blue-100 flex items-center justify-center gap-2 text-[10px] font-black uppercase active:scale-95 transition-all disabled:opacity-50">

              <RefreshCw size={14} className={isLoadingHistory ? "animate-spin" : ""} /> รีเฟรชข้อมูลจาก Google Sheet

            </button>


            {cloudError && (

              <div className="bg-rose-50 border border-rose-200 p-2 rounded-xl flex items-start gap-2">

                <AlertTriangle size={14} className="text-rose-500 mt-0.5 shrink-0" />

                <p className="text-[9px] text-rose-700 font-bold leading-tight">

                  กำลังแสดงข้อมูลออฟไลน์<br/>

                  <span className="font-normal opacity-80">แสดงผลจากข้อมูลในเครื่องล่าสุด (หากลบ/แก้ไข อาจยังไม่อัปเดตที่ Sheet)</span>

                </p>

              </div>

            )}

            

            {history.length === 0 && !isLoadingHistory ? <div className="text-center py-20 text-slate-300 font-black uppercase text-[9px] flex flex-col items-center gap-2"><History size={24} opacity={0.3}/> ไม่มีข้อมูลประวัติ</div> : 

              history.map((record) => (

                <div key={record.id} className="bg-white p-3 rounded-xl shadow-sm border border-slate-200 hover:border-slate-300 transition-colors">

                  <div className="flex justify-between items-center mb-2 pb-2 border-b border-slate-100">

                    <div className="flex items-center gap-2 font-black text-slate-800 text-[10px]"><Calendar size={12} className="text-orange-500"/> {formatDisplayDate(record.date)}</div>

                    <div className="flex gap-1">

                      {editingId === record.id ? (

                        <><button onClick={saveEdit} className="p-1.5 bg-emerald-100 text-emerald-600 rounded-md"><Check size={12}/></button><button onClick={() => setEditingId(null)} className="p-1.5 bg-slate-100 text-slate-500 rounded-md"><X size={12}/></button></>

                      ) : (

                        <>

                          <button onClick={() => {setEditingId(record.id); setEditBuffer(JSON.parse(JSON.stringify(record)));}} className="p-1.5 text-amber-500 bg-amber-50 rounded-md"><Pencil size={12}/></button>

                          <button onClick={() => deleteRecord(record)} className="p-1.5 text-rose-500 bg-rose-50 hover:bg-rose-100 hover:text-rose-600 rounded-md active:scale-90" title="ลบประวัติ"><Trash2 size={12}/></button>

                        </>

                      )}

                    </div>

                  </div>


                  {editingId === record.id ? (

                    <div className="space-y-2 p-1">

                      <div className="grid grid-cols-12 gap-1 text-[8px] font-black text-slate-400 uppercase text-center"><div className="col-span-4 text-left">รายการ</div><div className="col-span-4">เช้า</div><div className="col-span-4">เย็น</div></div>

                      {itemsList.map(item => (

                        <div key={item.id} className="grid grid-cols-12 items-center gap-1 border-b border-slate-50 pb-1">

                          <span className="col-span-4 text-[9px] font-black truncate">{item.label}</span>

                          <input type="number" value={editBuffer.stock[item.id].before} onChange={(e) => setEditBuffer(p => ({...p, stock: {...p.stock, [item.id]: {...p.stock[item.id], before: e.target.value}}}))} className="col-span-4 bg-slate-50 border rounded p-1 text-[10px] text-center font-bold" />

                          <input type="number" value={editBuffer.stock[item.id].after} onChange={(e) => setEditBuffer(p => ({...p, stock: {...p.stock, [item.id]: {...p.stock[item.id], after: e.target.value}}}))} className="col-span-4 bg-slate-50 border rounded p-1 text-[10px] text-center font-bold" />

                        </div>

                      ))}

                      <div className="grid grid-cols-4 gap-1 mt-2">

                        {salesTypes.map(s => (

                          <div key={s.id} className="flex flex-col gap-0.5">

                            <label className="text-[7px] font-bold text-slate-400 text-center truncate">{s.label}</label>

                            <input type="number" value={editBuffer.sales[s.id]} onChange={(e) => setEditBuffer(p => ({...p, sales: {...p.sales, [s.id]: e.target.value}}))} className="bg-orange-50 border border-orange-100 rounded p-1 text-[10px] text-center font-bold" />

                          </div>

                        ))}

                      </div>

                      <textarea value={editBuffer.note} onChange={(e) => setEditBuffer({...editBuffer, note: e.target.value})} className="w-full border rounded p-1.5 text-[10px] font-bold h-10 mt-2" placeholder="หมายเหตุ" />

                      <button onClick={saveEdit} className="w-full bg-emerald-600 text-white py-2 rounded-lg text-[10px] font-black mt-2">บันทึกการแก้ไขลง Sheet</button>

                    </div>

                  ) : (

                    <>

                      {record.sales && record.sales.total > 0 && (

                        <div className="mb-2 bg-slate-50 rounded-lg p-2 border border-slate-100 flex flex-wrap gap-1.5 items-center">

                          <BarChart3 size={10} className="text-slate-400" />

                          <span className="text-[9px] font-black text-slate-500 uppercase tracking-tighter">ขาย:</span>

                          {salesTypes.map(s => record.sales[s.id] > 0 && <span key={s.id} className="text-[8px] font-bold text-slate-700 bg-white px-1 rounded border">{s.label}:{record.sales[s.id]}</span>)}

                          <span className="text-[10px] font-black text-orange-600 ml-auto">รวม {record.sales.total}</span>

                        </div>

                      )}

                      <div className="grid grid-cols-1 sm:grid-cols-2 gap-x-3 gap-y-2">

                        {itemsList.map(item => {

                          const d = record.stock[item.id];

                          if (!d) return null;

                          const b = parseFloat(d.before) || 0; const a = parseFloat(d.after) || 0; const used = Math.max(0, b - a);

                          if (b === 0 && a === 0) return null;

                          return (

                            <div key={item.id} className="flex flex-col gap-0.5">

                              <div className="flex justify-between items-end"><span className={`text-[8px] font-black uppercase ${catColors[item.cat].split(' ')[1]}`}>{item.label}</span><div className="flex gap-1.5 text-[8px] font-bold"><span className="text-orange-600">ใช้ {used.toFixed(0)}</span><span className="text-slate-400">เหลือ {a.toFixed(0)}</span></div></div>

                              <div className="w-full h-1 bg-slate-100 rounded-full overflow-hidden border border-slate-200/50 flex">

                                <div style={{ width: `${b > 0 ? (used/b)*100 : 0}%` }} className={`h-full ${catColors[item.cat].split('bar:')[1]}`}></div>

                              </div>

                            </div>

                          );

                        })}

                      </div>

                      {record.note && <div className="mt-2 text-[8px] text-slate-400 font-bold italic line-clamp-1 border-l-2 border-slate-200 pl-1.5">"{record.note}"</div>}

                    </>

                  )}

                </div>

              ))

            }

          </div>

        )}

      </main>


      <div className="fixed bottom-2 left-1/2 -translate-x-1/2 px-4 pointer-events-none w-full max-w-xs text-center z-40">

        <div className="bg-slate-900/90 text-white text-[6px] py-1 px-4 rounded-full font-black tracking-widest uppercase shadow-lg inline-block border border-white/10 backdrop-blur-sm">KEBAB AI v16.2 - ROBUST FETCH</div>

      </div>

    </div>

  );

};
